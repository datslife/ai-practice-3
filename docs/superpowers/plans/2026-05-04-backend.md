# Backend Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Node.js + TypeScript backend with REST API, Socket.io real-time layer, PostgreSQL persistence, and Redis presence tracking.

**Architecture:** Express handles REST routes (`/auth`, `/users`, `/conversations`). Socket.io runs on the same HTTP server for real-time messaging and presence. PostgreSQL stores users, conversations, and messages. Redis tracks online presence with 30s TTL refreshed by client heartbeats.

**Tech Stack:** Node.js 20, TypeScript 5, Express 4, pg, ioredis, socket.io 4, jsonwebtoken, bcryptjs, uuid, Jest + ts-jest + supertest, socket.io-client (test)

```typescript
import jwt from 'jsonwebtoken';
import { JwtPayload } from '../types';

function secret(): string {
  const s = process.env.JWT_SECRET;
  if (!s) throw new Error('JWT_SECRET not set');
  return s;
}

export function signToken(payload: JwtPayload): string {
  return jwt.sign(payload, secret(), { expiresIn: '7d' });
}

export function verifyToken(token: string): JwtPayload {
  return jwt.verify(token, secret()) as JwtPayload;
}
```

```typescript
import { Request, Response, NextFunction } from 'express';
import { verifyToken } from './jwt';

export function requireAuth(req: Request, res: Response, next: NextFunction): void {
  const auth = req.headers.authorization;
  if (!auth?.startsWith('Bearer ')) {
    res.status(401).json({ error: 'Unauthorized' });
    return;
  }
  try {
    req.user = verifyToken(auth.slice(7));
    next();
  } catch {
    res.status(401).json({ error: 'Unauthorized' });
  }
}
```

```typescript
import bcrypt from 'bcryptjs';
import { v4 as uuidv4 } from 'uuid';
import { pool } from '../db/client';
import { signToken } from './jwt';
import { User } from '../types';

export async function register(
  email: string,
  name: string,
  password: string
): Promise<{ user: User; token: string }> {
  const hash = await bcrypt.hash(password, 10);
  const { rows } = await pool.query<User>(
    `INSERT INTO users (id, email, name, password_hash, created_at)
     VALUES ($1, $2, $3, $4, NOW())
     RETURNING id, email, name, created_at`,
    [uuidv4(), email.toLowerCase().trim(), name.trim(), hash]
  );
  const user = rows[0];
  return { user, token: signToken({ userId: user.id, email: user.email }) };
}

export async function login(
  email: string,
  password: string
): Promise<{ user: User; token: string }> {
  const { rows } = await pool.query<User & { password_hash: string }>(
    `SELECT id, email, name, created_at, password_hash
     FROM users WHERE email = $1`,
    [email.toLowerCase().trim()]
  );
  const row = rows[0];
  if (!row) throw new Error('INVALID_CREDENTIALS');
  const valid = await bcrypt.compare(password, row.password_hash);
  if (!valid) throw new Error('INVALID_CREDENTIALS');
  const { password_hash: _omit, ...user } = row;
  return { user, token: signToken({ userId: user.id, email: user.email }) };
}
```

```typescript
import { Router } from 'express';
import { register, login } from './service';

const router = Router();

router.post('/register', async (req, res) => {
  const { email, name, password } = req.body as Record<string, unknown>;
  if (typeof email !== 'string' || typeof name !== 'string' || typeof password !== 'string') {
    res.status(400).json({ error: 'email, name, and password are required' });
    return;
  }
  if (!email.trim() || !name.trim() || !password) {
    res.status(400).json({ error: 'email, name, and password must not be empty' });
    return;
  }
  try {
    const result = await register(email, name, password);
    res.status(201).json(result);
  } catch (err: unknown) {
    const pgErr = err as { code?: string };
    if (pgErr.code === '23505') {
      res.status(409).json({ error: 'Email already in use' });
      return;
    }
    res.status(500).json({ error: 'Registration failed' });
  }
});

router.post('/login', async (req, res) => {
  const { email, password } = req.body as Record<string, unknown>;
  if (typeof email !== 'string' || typeof password !== 'string') {
    res.status(400).json({ error: 'email and password are required' });
    return;
  }
  try {
    const result = await login(email, password);
    res.json(result);
  } catch {
    res.status(401).json({ error: 'Invalid credentials' });
  }
});

export default router;
```

- [ ] **Step 4: Write `backend/src/presence/service.ts`**

```typescript
import { redis } from '../redis/client';

const PRESENCE_TTL = 30;

function presenceKey(userId: string): string {
  return `presence:${userId}`;
}

export async function setOnline(userId: string): Promise<void> {
  await redis.set(presenceKey(userId), 'online', 'EX', PRESENCE_TTL);
}

export async function setOffline(userId: string): Promise<void> {
  await redis.del(presenceKey(userId));
}

export async function refreshTTL(userId: string): Promise<void> {
  await redis.expire(presenceKey(userId), PRESENCE_TTL);
}

export async function getStatusMap(
  userIds: string[]
): Promise<Record<string, 'online' | 'offline'>> {
  if (userIds.length === 0) return {};
  const pipeline = redis.pipeline();
  userIds.forEach((id) => pipeline.get(presenceKey(id)));
```

```typescript
import { v4 as uuidv4 } from 'uuid';
import { pool } from '../db/client';
import { Conversation, Message } from '../types';

export async function getOrCreateConversation(
  userAId: string,
  userBId: string
): Promise<Conversation> {
  const [aId, bId] = [userAId, userBId].sort();
  const { rows } = await pool.query<Conversation>(
    `INSERT INTO conversations (id, user_a_id, user_b_id, created_at)
     VALUES ($1, $2, $3, NOW())
     ON CONFLICT ON CONSTRAINT conversations_user_pair_unique
     DO UPDATE SET created_at = conversations.created_at
     RETURNING *`,
    [uuidv4(), aId, bId]
  );
  return rows[0];
}

export async function createMessage(
  conversationId: string,
  senderId: string,
  content: string
): Promise<Message> {
  const { rows } = await pool.query<Message>(
    `INSERT INTO messages (id, conversation_id, sender_id, content, created_at)
     VALUES ($1, $2, $3, $4, NOW()) RETURNING *`,
    [uuidv4(), conversationId, senderId, content]
  );
  return rows[0];
}

export async function markMessageRead(messageId: string): Promise<Message | null> {
  const { rows } = await pool.query<Message>(
    `UPDATE messages SET read_at = NOW()
     WHERE id = $1 AND read_at IS NULL RETURNING *`,
    [messageId]
  );
  return rows[0] ?? null;
}

export async function listMessages(
  conversationId: string,
  limit = 50
): Promise<Message[]> {
  const { rows } = await pool.query<Message>(
    `SELECT * FROM messages WHERE conversation_id = $1
     ORDER BY created_at ASC LIMIT $2`,
    [conversationId, limit]
  );
  return rows;
}

export async function getConversationForUsers(
  userId: string,
  conversationId: string
): Promise<Conversation | null> {
  const { rows } = await pool.query<Conversation>(
    `SELECT * FROM conversations
     WHERE id = $1 AND (user_a_id = $2 OR user_b_id = $2)`,
    [conversationId, userId]
  );
  return rows[0] ?? null;
}
```

```typescript
import { Server as HttpServer } from 'http';
import { Server as SocketServer, Socket } from 'socket.io';
import { verifyToken } from '../auth/jwt';
import { setOnline, setOffline, refreshTTL } from '../presence/service';
import {
  createMessage,
  markMessageRead,
  getOrCreateConversation,
} from '../messages/service';
import { JwtPayload } from '../types';

type AuthSocket = Socket & { user: JwtPayload };

export function createSocketServer(httpServer: HttpServer): SocketServer {
  const io = new SocketServer(httpServer, {
    cors: { origin: '*' },
  });

  io.use((socket, next) => {
    const token = (socket.handshake.auth as Record<string, unknown>)?.token;
    if (typeof token !== 'string' || !token) {
      return next(new Error('AUTH_REQUIRED'));
    }
    try {
      (socket as AuthSocket).user = verifyToken(token);
      next();
    } catch {
      next(new Error('AUTH_INVALID'));
    }
  });

  io.on('connection', async (socket) => {
    const { user } = socket as AuthSocket;

    await setOnline(user.userId);
    io.emit('presence:update', { userId: user.userId, status: 'online' });

    socket.on('heartbeat', async () => {
      await refreshTTL(user.userId).catch(() => null);
    });

    socket.on(
      'message:send',
      async ({
        recipientId,
        content,
        tempId,
      }: {
        recipientId: string;
        content: string;
        tempId: string;
      }) => {
        if (!content?.trim()) {
          socket.emit('message:error', { tempId, error: 'EMPTY_CONTENT' });
          return;
        }
        try {
          const conv = await getOrCreateConversation(user.userId, recipientId);
          const message = await createMessage(conv.id, user.userId, content.trim());
          socket.emit('message:sent', {
            tempId,
            id: message.id,
            conversationId: conv.id,
            createdAt: message.created_at,
          });
          const allSockets = await io.fetchSockets();
          const recipientSocket = allSockets.find(
            (s) => (s as unknown as AuthSocket).user?.userId === recipientId
          );
          if (recipientSocket) {
            recipientSocket.emit('message:new', {
              id: message.id,
              senderId: user.userId,
              content: message.content,
              conversationId: conv.id,
              createdAt: message.created_at,
            });
          }
        } catch {
          socket.emit('message:error', { tempId, error: 'SEND_FAILED' });
        }
      }
    );

    socket.on('message:read', async ({ messageId }: { messageId: string }) => {
      const message = await markMessageRead(messageId).catch(() => null);
      if (!message) return;
      const allSockets = await io.fetchSockets();
      const senderSocket = allSockets.find(
        (s) => (s as unknown as AuthSocket).user?.userId === message.sender_id
      );
      if (senderSocket) {
        senderSocket.emit('message:read', {
          messageId,
          readAt: message.read_at,
        });
      }
    });

    socket.on('disconnect', async () => {
      await setOffline(user.userId).catch(() => null);
      io.emit('presence:update', { userId: user.userId, status: 'offline' });
    });
  });

  return io;
}
```
